\subsection{Système de review}
\frame{
  \frametitle{Modèle pilote/copilote}
  \begin{itemize}
    \item Groupe divisé en deux binômes
    \item Un membre code, l'autre observe, conseille, aide
    \item Intérêt : parallélisation des tâches, avec partage de connaissances
  \end{itemize}
}

\frame{
  \frametitle{Les différentes branches}
  \begin{itemize}
    \item Master : dernière version stable du projet
    \item Develop : version en cours de développement
    \item Une branche par fonctionnalité
    \item Une fois revues et validées, les branches de fonctionnalité fusionnent avec develop
    \item Une fois achevée, revue et validée, les versions sont taggées dans master 
  \end{itemize}
}
